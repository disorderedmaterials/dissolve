  # -- Partial g(r) (unbound terms)
  Module  DataTest  'Partial g(r) (unbound)'
    Threshold  0.06
    ErrorType  RMSE
    Data1D  'Bulk//OriginalGR//OW-OW//Unbound'  xy  'epsr/water.EPSR.g01'

  # -- Partial S(Q) (unbound terms)
